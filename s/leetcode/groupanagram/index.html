49. Group Anagrams

Given an array of strings, group anagrams together.

For example, given: ["eat", "tea", "tan", "ate", "nat", "bat"], Return:

[
  ["ate", "eat","tea"],
  ["nat","tan"],
  ["bat"]
]

Note: For the return value, each inner list’s elements must follow the lexicographic order. All inputs will be in lower-case.

  • 解一：使用前面#242發展的演算法，用兩個index : i, j, 第一個字的時候，掃後面的字是不是anagram，一樣的就放進同一個字的arraylist, 此法一直過不了時間限制
public static boolean isAnagram2(String s, String t){
	char[] scs = s.replaceAll("\\s", "").toCharArray();
	char[] tcs = t.replaceAll("\\s", "").toCharArray();
	Arrays.sort(scs);
	Arrays.sort(tcs);
	if(scs.length!=tcs.length){
		return false;
	}else{
		for(int i=0; i<scs.length; i++){
			if(scs[i]!=tcs[i]){
				return  false;
			}
		}
		return true;
	}
}

public List<List<String>> groupAnagrams(String[] strs) {
    List<List<String>> result = new ArrayList<List<String>>();
	for(int i=0; i<strs.length;i++){
		List<String> inner = new ArrayList<String>();
		if(strs[i]!=""){
			inner.add(strs[i]);
			for(int j=0 ;j<strs.length;j++){
				if(i!=j && isAnagram2(strs[i],strs[j])){
					inner.add(strs[j]);
					strs[j] = "";
				}
			}
			result.add(inner);
		}
	}
	return result;
}
  • 解二：把字先sort過，放進map, 如果已經存在了，代表有anagram了，把自己的index存進value的list即可。
public List<List<String>> groupAnagrams(String[] strs) {
    List<List<String>> result = new ArrayList<List<String>>();
	if(strs.length==0){
		return result;
	}
	Map<String, List<Integer>> table = new HashMap<String, List<Integer>>();
	for(int i=0;i<strs.length;i++){
		char[] c = strs[i].replaceAll("\\s", "").toCharArray();
		Arrays.sort(c);
		String tmpSorted = new String(c);
		List<Integer> index = table.get(tmpSorted);
		if(index == null){
			index = new ArrayList<Integer>();
			index.add(i);
			table.put(tmpSorted, index);
		}else{
			index.add(i);
		}
	}
	for(String k : table.keySet()){
		List<String> inner = new ArrayList<String>();
		List<Integer> idx = table.get(k);
		for(Integer i : idx){
			inner.add(strs[i]);
		}
		Collections.sort(inner);
		result.add(inner);
	}
	return result;
}
